T -> S
S -> [expr] S'
S' -> [expr] S' | [T_ASSIGN T_ID expr'] | e

expr -> intbinop intexpr intexpr | intunop intexpr | floatbinop floatexpr floatexpr | floatunop floatexpr | 
    T_PLUS strexpr strexpr | boolbinop boolexpr boolexpr | boolnumbinop intexpr intexpr | 
    boolnumbinop floatexpr floatexpr | boolstrbinop strexpr strexp | T_NOT boolexpr | type T_ID | stmt

intbinop -> T_PLUS | T_MINUS | T_MULT | T_DIV | T_MOD | T_EXP
intunop -> T_MINUS | T_SIN | T_COS | T_TAN
intexpr -> T_INT | T_ID | [intexpr]

floatbinop -> T_PLUS | T_MINUS | T_MULT | T_DIV
floatbinop -> T_MINUS | T_SIN | T_COS | T_TAN
floatexpr -> T_FLOAT | T_ID | [floatexpr]

boolbinop -> T_EQ | T_NOTEQ | T_AND | T_OR
boolexpr -> T_BOOL | T_ID | [boolexpr]

boolnumbinop -> T_EQ | T_NOTEQ | T_LT | T_LTEQ | T_GT | T_GTEQ

boolstrbinop -> T_EQ | T_NOTEQ

strexpr -> T_STRING | T_ID | [strexpr]

exprlist -> expr' | expr' exprlist

varlist -> [type T_ID] | [type T_ID] varlist

stmt -> ifstmt | whilestmt | letstmt | printstmt
ifstmt -> T_IF expr' expr' expr' | T_IF expr' expr'
whilestmt -> T_WHILE expr' exprlist
letstmt -> T_LET varlist
printstmt -> T_STDOUT expr'

type -> T_STRINGTYPE | T_INTTYPE | T_FLOATTYPE | T_BOOLTYPE
