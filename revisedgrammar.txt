T -> S
S -> [expr] S'
S' -> [expr] S' | [T_ASSIGN T_ID expr'] | e
expr -> binop expr' expr' | unop expr' expr' | type T_ID | stmt
expr' -> const | T_ID | [expr]

exprlist -> expr' | expr' exprlist

varlist -> [type T_ID] | [type T_ID] varlist

stmt -> ifstmt | whilestmt | letstmt | printstmt
ifstmt -> T_IF expr' expr' expr' | T_IF expr' expr'
whilestmt -> T_WHILE expr' exprlist
letstmt -> T_LET varlist
printstmt -> T_STDOUT expr'

const -> T_CONSTSTR | T_FLOAT | T_INT | T_BOOL

type -> T_STRINGTYPE | T_INTTYPE | T_FLOATTYPE | T_BOOLTYPE

binop -> T_PLUS | T_MINUS | T_MULT | T_DIV | T_MOD | T_NOT | T_EQ | T_LT | T_LTEQ | T_GT | T_GTEQ | T_NOTEQ | T_OR T| T_AND

unop -> T_MINUS | T_NOT | T_SIN | T_COS | T_TAN
