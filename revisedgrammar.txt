# all brackets in this document correspond to T_LBRACKET / T_RBRACKET etc.
T -> S

S -> expr S | expr

expr-> [binop expr' expr'] | [unop expr'] | [T_ASSIGN T_ID expr']
expr' -> [binop expr' expr'] | [unop expr'] | const | T_ID

binop -> T_PLUS | T_MINUS | T_MULT | T_DIV | T_MOD | T_EXP
unop -> T_NOT | T_SIN | T_COS | T_TAN | T_MINUS

const -> T_FLOAT | T_BOOL | T_INT | T_CONSTSTR

exprlist -> expr | expr exprlist

varlist -> [type T_ID] | [type T_ID] varlist

stmt -> [ifstmt] | [whilestmt] | [letstmt] | [printstmt] | [assignstmt]

# if cond then doexpr else dootherexpr
# if cond then doexpr
ifstmt -> T_IF expr expr expr | T_IF expr expr

whilestmt -> T_WHILE expr exprlist
letstmt -> T_LET varlist
printstmt -> T_STDOUT expr
assignstmt -> T_ASSIGN T_ID expr

type -> T_STRINGTYPE | T_INTTYPE | T_FLOATTYPE | T_BOOLTYPE
